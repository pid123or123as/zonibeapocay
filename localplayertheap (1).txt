-- ============================================================
-- Helpers
-- ============================================================
local function getCharacterData()
    local character = LocalPlayerObj and LocalPlayerObj.Character
    if not character then return nil, nil end
    return character:FindFirstChild("Humanoid"), character:FindFirstChild("HumanoidRootPart")
end

local function isCharacterValid(humanoid, rootPart)
    return humanoid and rootPart and humanoid.Health > 0
end

local function isInVehicle(rootPart)
    local part = rootPart
    while part do
        if part:IsA("Seat") or part:IsA("VehicleSeat") then return true end
        part = part.Parent
    end
    return false
end

local function isInputFocused()
    return Services and Services.UserInputService and
           Services.UserInputService:GetFocusedTextBox() ~= nil
end

local function getCustomMoveDirection()
    if not Services.UserInputService or not Services.Workspace.CurrentCamera then
        SpeedStatus.CurrentMoveDirection = Vector3.new(0, 0, 0)
        return Vector3.new(0, 0, 0)
    end

    local cam      = Services.Workspace.CurrentCamera.CFrame
    local fwd      = Vector3.new(cam.LookVector.X,  0, cam.LookVector.Z)
    local right    = Vector3.new(cam.RightVector.X, 0, cam.RightVector.Z)
    if fwd.Magnitude == 0 or right.Magnitude == 0 then
        SpeedStatus.CurrentMoveDirection = Vector3.new(0, 0, 0)
        return Vector3.new(0, 0, 0)
    end
    fwd   = fwd.Unit
    right = right.Unit

    local uis = Services.UserInputService
    local w = uis:IsKeyDown(Enum.KeyCode.W) and  1 or 0
    local s = uis:IsKeyDown(Enum.KeyCode.S) and -1 or 0
    local a = uis:IsKeyDown(Enum.KeyCode.A) and -1 or 0
    local d = uis:IsKeyDown(Enum.KeyCode.D) and  1 or 0

    local input = Vector3.new(a + d, 0, w + s)
    if input.Magnitude > 0 then
        input = input.Unit
        local target = fwd * input.Z + right * input.X
        if target.Magnitude > 0 then target = target.Unit end
        local alpha = SpeedStatus.SmoothnessFactor
        SpeedStatus.CurrentMoveDirection =
            SpeedStatus.CurrentMoveDirection * (1 - alpha) + target * alpha
        if SpeedStatus.CurrentMoveDirection.Magnitude > 0 then
            SpeedStatus.CurrentMoveDirection = SpeedStatus.CurrentMoveDirection.Unit
        end
    else
        SpeedStatus.CurrentMoveDirection = Vector3.new(0, 0, 0)
    end
    return SpeedStatus.CurrentMoveDirection
end

-- ============================================================
-- Timer
-- ============================================================
local Timer = {}

Timer.Start = function()
    if TimerStatus.Running or not Services then return end
    local ok = pcall(function()
        setfflag("SimEnableStepPhysics", "True")
        setfflag("SimEnableStepPhysicsSelective", "True")
    end)
    if not ok then
        notify("Timer", "Failed to enable physics simulation.", true)
        return
    end
    TimerStatus.Running = true
    TimerStatus.Connection = Services.RunService.RenderStepped:Connect(function(dt)
        if not TimerStatus.Enabled or TimerStatus.Speed <= 1 then return end
        local humanoid, rootPart = getCharacterData()
        if not isCharacterValid(humanoid, rootPart) then return end
        local stepOk, err = pcall(function()
            Services.RunService:Pause()
            Services.Workspace:StepPhysics(dt * (TimerStatus.Speed - 1), {rootPart})
            Services.RunService:Run()
        end)
        if not stepOk then
            warn("Timer physics step failed: " .. tostring(err))
            Timer.Stop()
            notify("Timer", "Physics step failed. Timer stopped.", true)
        end
    end)
    notify("Timer", "Started with speed: " .. TimerStatus.Speed, true)
end

Timer.Stop = function()
    if TimerStatus.Connection then
        TimerStatus.Connection:Disconnect()
        TimerStatus.Connection = nil
    end
    TimerStatus.Running = false
    notify("Timer", "Stopped", true)
end

Timer.SetSpeed = function(newSpeed)
    TimerStatus.Speed = math.clamp(newSpeed, 1, 15)
    MovementEnhancements.Config.Timer.Speed = TimerStatus.Speed
end

-- ============================================================
-- Disabler
-- ============================================================
local Disabler = {}

Disabler.DisableSignals = function(character)
    if not character then return end
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end
    for _, conn in ipairs(getconnections(rootPart:GetPropertyChangedSignal("CFrame"))) do
        pcall(function() hookfunction(conn.Function, newcclosure(function() end)) end)
    end
    for _, conn in ipairs(getconnections(rootPart:GetPropertyChangedSignal("Velocity"))) do
        pcall(function() hookfunction(conn.Function, newcclosure(function() end)) end)
    end
end

Disabler.Start = function()
    if DisablerStatus.Running or not LocalPlayerObj then return end
    DisablerStatus.Running = true
    DisablerStatus.Connection = LocalPlayerObj.CharacterAdded:Connect(Disabler.DisableSignals)
    if LocalPlayerObj.Character then
        Disabler.DisableSignals(LocalPlayerObj.Character)
    end
    notify("Disabler", "Started", true)
end

Disabler.Stop = function()
    if DisablerStatus.Connection then
        DisablerStatus.Connection:Disconnect()
        DisablerStatus.Connection = nil
    end
    DisablerStatus.Running = false
    notify("Disabler", "Stopped", true)
end

-- Speed
-- ============================================================
local Speed = {}

Speed.UpdateMovement = function(humanoid, rootPart, dir, currentTime, dt)
    if not isCharacterValid(humanoid, rootPart) then return end
    if SpeedStatus.Method == "CFrame" then
        if dir.Magnitude > 0 then
            local newPos = rootPart.CFrame + (dir * SpeedStatus.Speed * dt)
            rootPart.CFrame = CFrame.new(newPos.Position, newPos.Position + dir)
        end
    elseif SpeedStatus.Method == "PulseTP" then
        if dir.Magnitude > 0 and currentTime - SpeedStatus.LastPulseTPTime >= SpeedStatus.PulseTPFrequency then
            local scaledDist = SpeedStatus.PulseTPDistance * (SpeedStatus.Speed / 16)
            local tpVec      = dir.Unit * scaledDist
            local dest       = rootPart.Position + tpVec
            local rp         = RaycastParams.new()
            rp.FilterDescendantsInstances = {LocalPlayerObj.Character}
            rp.FilterType = Enum.RaycastFilterType.Exclude
            if not Services.Workspace:Raycast(rootPart.Position, tpVec, rp) then
                rootPart.CFrame = CFrame.new(dest, dest + dir)
                SpeedStatus.LastPulseTPTime = currentTime
            end
        end
    end
end

Speed.UpdateJumps = function(humanoid, rootPart, currentTime, dir)
    if not isCharacterValid(humanoid, rootPart) then return end
    if SpeedStatus.AutoJump and dir.Magnitude > 0 and
       currentTime - SpeedStatus.LastJumpTime >= SpeedStatus.JumpInterval then
        local state = humanoid:GetState()
        if state == Enum.HumanoidStateType.Running or state == Enum.HumanoidStateType.Landed then
            humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            SpeedStatus.LastJumpTime = currentTime
        end
    end
end

Speed.Start = function()
    if SpeedStatus.Running or not Services then return end
    SpeedStatus.Running = true
    SpeedStatus.Connection = Services.RunService.Heartbeat:Connect(function(dt)
        if not SpeedStatus.Enabled then SpeedStatus.Running = false; return end
        local humanoid, rootPart = getCharacterData()
        if not isCharacterValid(humanoid, rootPart) then return end
        local now = tick()
        local dir = getCustomMoveDirection()
        Speed.UpdateMovement(humanoid, rootPart, dir, now, dt)
        Speed.UpdateJumps(humanoid, rootPart, now, dir)
    end)
    notify("Speed", "Started with Method: " .. SpeedStatus.Method, true)
end

Speed.Stop = function()
    if SpeedStatus.Connection then
        SpeedStatus.Connection:Disconnect()
        SpeedStatus.Connection = nil
    end
    SpeedStatus.Running = false
    notify("Speed", "Stopped", true)
end

Speed.SetSpeed = function(v)
    SpeedStatus.Speed = math.clamp(v, 16, 250)
    MovementEnhancements.Config.Speed.Speed = SpeedStatus.Speed
end

Speed.SetMethod = function(v)
    SpeedStatus.Method = v
    MovementEnhancements.Config.Speed.Method = v
    if SpeedStatus.Running then Speed.Stop(); Speed.Start() end
end

Speed.SetPulseTPDistance = function(v)
    SpeedStatus.PulseTPDistance = math.clamp(v, 1, 20)
    MovementEnhancements.Config.Speed.PulseTPDist = SpeedStatus.PulseTPDistance
end

Speed.SetPulseTPFrequency = function(v)
    SpeedStatus.PulseTPFrequency = math.clamp(v, 0.1, 2)
    MovementEnhancements.Config.Speed.PulseTPDelay = SpeedStatus.PulseTPFrequency
end

Speed.SetSmoothnessFactor = function(v)
    SpeedStatus.SmoothnessFactor = math.clamp(v, 0, 1)
    MovementEnhancements.Config.Speed.SmoothnessFactor = SpeedStatus.SmoothnessFactor
end

-- ============================================================
-- Fly
-- ============================================================
local Fly = {}

Fly.Start = function()
    if FlyStatus.Running or not Services or not Services.Workspace.CurrentCamera then return end
    local humanoid, rootPart = getCharacterData()
    if not isCharacterValid(humanoid, rootPart) or isInVehicle(rootPart) then return end
    FlyStatus.Running = true
    FlyStatus.BodyVelocity = Instance.new("BodyVelocity")
    FlyStatus.BodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    FlyStatus.BodyVelocity.Parent   = rootPart
    FlyStatus.Connection = Services.RunService.Heartbeat:Connect(function(dt)
        if not FlyStatus.Enabled then FlyStatus.Running = false; return end
        local now = tick()
        if now - FlyStatus.LastUpdateTime < MovementEnhancements.Config.Fly.UpdateInterval then return end
        FlyStatus.LastUpdateTime = now

        local dir      = getCustomMoveDirection()
        local velocity = dir * FlyStatus.Speed
        local vertVel  = 0

        local upKey, downKey = FlyStatus.VerticalKeys:match("(.+)/(.+)")
        if upKey and downKey then
            local ok, err = pcall(function()
                local uis = Services.UserInputService
                if uis:IsKeyDown(Enum.KeyCode[upKey]) then
                    vertVel = FlyStatus.VerticalSpeed
                elseif uis:IsKeyDown(Enum.KeyCode[downKey]) then
                    vertVel = -FlyStatus.VerticalSpeed
                end
            end)
            if not ok then
                notify("Fly", "Invalid vertical keys. " .. tostring(err), true)
                return
            end
        end

        FlyStatus.BodyVelocity.Velocity = Vector3.new(velocity.X, vertVel, velocity.Z)

        if dir.Magnitude > 0 or vertVel ~= 0 then
            local cam = Services.Workspace.CurrentCamera
            if cam then
                local look = dir.Magnitude > 0 and dir or cam.CFrame.LookVector
                rootPart.CFrame = CFrame.new(rootPart.Position, rootPart.Position + look)
            end
        end
    end)
    notify("Fly", "Started with Speed: " .. FlyStatus.Speed, true)
end

Fly.Stop = function()
    if FlyStatus.Connection then
        FlyStatus.Connection:Disconnect()
        FlyStatus.Connection = nil
    end
    if FlyStatus.BodyVelocity then
        FlyStatus.BodyVelocity:Destroy()
        FlyStatus.BodyVelocity = nil
    end
    FlyStatus.Running = false
    notify("Fly", "Stopped", true)
end

Fly.SetSpeed = function(v)
    FlyStatus.Speed = math.clamp(v, 10, 200)
    MovementEnhancements.Config.Fly.Speed = FlyStatus.Speed
end

Fly.SetVerticalSpeed = function(v)
    FlyStatus.VerticalSpeed = math.clamp(v, 10, 200)
    MovementEnhancements.Config.Fly.VerticalSpeed = FlyStatus.VerticalSpeed
end

Fly.SetVerticalKeys = function(v)
    FlyStatus.VerticalKeys = v
    MovementEnhancements.Config.Fly.VerticalKeys = v
end

-- ============================================================
-- Keybind helper: правильный паттерн по MacLib скиллу
-- Callback  → FAB-нажатие / MacLib trigger
-- onBinded  → сохраняем KeyCode в Config, подключаем InputBegan
-- InputBegan → fallback для PC (KeyCode-нажатия мимо MacLib)
-- ============================================================
local InputBeganConnections = {}

local function setupKeybind(keybindElement, configKeyRef, action)
    -- Отключаем старый InputBegan если есть
    if InputBeganConnections[configKeyRef] then
        InputBeganConnections[configKeyRef]:Disconnect()
        InputBeganConnections[configKeyRef] = nil
    end

    keybindElement.onBinded = function(bind)
        -- Сохраняем актуальный KeyCode
        configKeyRef[1] = bind

        -- Переподключаем InputBegan с новым ключом
        if InputBeganConnections[configKeyRef] then
            InputBeganConnections[configKeyRef]:Disconnect()
        end
        if bind then
            InputBeganConnections[configKeyRef] =
                Services.UserInputService.InputBegan:Connect(function(input, gp)
                    if gp or isInputFocused() then return end
                    if input.KeyCode == bind then
                        action()
                    end
                end)
        end
    end
end

-- ============================================================
-- Init
-- ============================================================
function MovementEnhancements.Init(UI, coreParam, notifyFunc)
    core         = coreParam
    Services     = core.Services
    PlayerData   = core.PlayerData
    notify       = notifyFunc
    LocalPlayerObj = PlayerData.LocalPlayer

    if LocalPlayerObj then
        LocalPlayerObj.CharacterAdded:Connect(function(newChar)
            if DisablerStatus.Enabled then Disabler.DisableSignals(newChar) end
            if SpeedStatus.Enabled    then Speed.Start() end
            if FlyStatus.Enabled      then Fly.Start()   end
        end)
    end

    SetupUI(UI)
end

-- ============================================================
-- Destroy
-- ============================================================
function MovementEnhancements:Destroy()
    local connections = {
        TimerStatus.Connection,
        DisablerStatus.Connection,
        SpeedStatus.Connection,
        FlyStatus.Connection
    }
    for _, conn in ipairs(connections) do
        if conn then conn:Disconnect() end
    end
    TimerStatus.Connection    = nil
    DisablerStatus.Connection = nil
    SpeedStatus.Connection    = nil
    FlyStatus.Connection      = nil

    for _, conn in pairs(InputBeganConnections) do
        if conn then conn:Disconnect() end
    end
    InputBeganConnections = {}

    if FlyStatus.BodyVelocity then
        FlyStatus.BodyVelocity:Destroy()
        FlyStatus.BodyVelocity = nil
    end

    TimerStatus.Running    = false
    DisablerStatus.Running = false
    SpeedStatus.Running    = false
    FlyStatus.Running      = false
end

return MovementEnhancements
